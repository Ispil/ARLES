#ROMAN PRAETORIANI
arles_retinue_roman_praetorian_create_effect = {
	primary_title = {
		create_title = {
			tier = COUNT
			landless = yes
			custom_created = yes
			culture = roman
			name = RETINUE_ROMAN_PRAETORIAN
			holder = FROM
			base_title = THIS
			mercenary = yes
			replace_captain_on_death = yes
			army_template = roman_retinue_praetorian_composition
			ruler = "CAPTAIN_ROMAN_PRAETORIAN"
			ruler_female = "CAPTAIN_ROMAN_PRAETORIAN_FEMALE"
			foa = "CAPTAIN_FOA_ROMAN_PRAETORIAN"
		}
		new_title = {
			set_title_flag = arles_dyn_retinue
			set_title_flag = heavy_infantry_retinue
		}
		FROM = {
			set_defacto_liege = ROOT
			set_character_flag = is_retinue_commander
		}
	}
}
